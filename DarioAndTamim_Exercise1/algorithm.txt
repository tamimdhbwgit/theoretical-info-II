Algorithm CalculateFactorial(n)
Input: An integer n (where n >= 0)
Output: The factorial of n (n!)

    // Handle edge cases for 0! and negative numbers
    if n < 0 then
        return "Error: Factorial not defined for negative numbers"
    if n == 0 then
        return 1

    // Initialize the result to 1
    result = 1

    // Loop from 1 up to n, multiplying the result by the counter
    for i = 1 to n do
        result = result * i

    return result


// Exercise-1 no:6:
Algorithm CountTotalVowels(text)
Input: A string of characters 'text'
Output: The total number of vowels in the string

    // Initialize a single counter for all vowels to 0
    vowelCount = 0

    // Loop through every character in the string
    for i = 0 to length of text - 1 do
        
        // Extract the current character and convert it to lowercase
        c = to_lowercase(text[i])

        // Check if the character is any of the five vowels
        if c == 'a' or c == 'e' or c == 'i' or c == 'o' or c == 'u' then
            // If it is a vowel, increase the total count by 1
            vowelCount = vowelCount + 1

    // Return the final total count
    return vowelCount
